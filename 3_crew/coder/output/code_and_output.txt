# Código: leibniz_pi.py
import time
import math

N = 10000

start = time.time()

s = 0.0
sign = 1.0
den = 1.0
for k in range(N):
    s += sign / den
    sign = -sign
    den += 2.0

pi_approx = 4.0 * s
end = time.time()

print("Approximation of pi using 10,000 terms of Leibniz series:")
print(f"pi_approx = {pi_approx:.15f}")
print(f"math.pi   = {math.pi:.15f}")
print(f"abs error = {abs(pi_approx - math.pi):.15f}")
print(f"terms = {N}")
print(f"time elapsed = {end - start:.6f} seconds")
